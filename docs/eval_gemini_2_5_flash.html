SELECT partner,
  ROUND((SUM(success_trx)::numeric / NULLIF(SUM(total_trx), 0)) * 100, 2) AS success_rate_pct,
  SUM(total_trx) AS total_transaksi
FROM daily_master
WHERE date >= '2026-05-01' AND date <= '2026-05-31'
GROUP BY partner
ORDER BY success_rate_pct DESC
